	//ru
	Анализатор = Новый("AddIn.Parser.BSL");
	
	Текст = "// тест ком
	|Если Истина Тогда 
	|	в = 1;
	|КонецЕсли;";
	Текст_en = "// test 
	|if true then
	|	a = 1;
	|elsif false then
	|	b = 0;
	|	a = 1;
	|endif;";
	Анализатор.Текст = Текст;
	Анализатор.Выполнить(-1);
	ПолучитьДерево(Анализатор, Строки);
	
	Анализатор.Выполнить(2);
	Пока Анализатор.Следующий() Цикл
		НоваяСтрока = Каналы.Добавить();
		НоваяСтрока.Тип = Анализатор.Тип();
		НоваяСтрока.Текст = Анализатор.Текст();
		НоваяСтрока.НомерСтроки = Анализатор.НомерСтроки();
		НоваяСтрока.НомерСимвола = Анализатор.НомерСимвола();	
	КонецЦикла;
	
КонецЕсли;

Процедура ПолучитьДерево(Анализатор, Строки)
	
	Пока Анализатор.Следующий() Цикл 
		
		НоваяСтрока = Строки.Добавить();
		НоваяСтрока.Тип = Анализатор.Тип();
		
		Если Анализатор.ЭтоЛексема() Тогда 
			НоваяСтрока.Текст = Анализатор.Текст();
			НоваяСтрока.НомерСтроки = Анализатор.НомерСтроки();
			НоваяСтрока.НомерСимвола = Анализатор.НомерСимвола();
		Иначе
			Если Анализатор.ПолучитьЭлементы() Тогда 
				ПолучитьДерево(Анализатор, НоваяСтрока.ПолучитьЭлементы());
			Иначе 
				ВызватьИсключение Неопределено;
			КонецЕсли;
		КонецЕсли;
		
	КонецЦикла;
	
КонецПроцедуры
